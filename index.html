function calcScores() {

  let cur = 0, proj = 0;

  const cats = {};

  questions.forEach((q, qi) => {

    const sel = answers[qi];

    if (sel === undefined) return;

    const o = q.opts[sel];

    cur += o.pts; proj += o.proj;

    if (!cats[q.cat]) cats[q.cat] = { cur: 0, proj: 0, max: 0 };

    cats[q.cat].cur += o.pts;

    cats[q.cat].proj += o.proj;

    cats[q.cat].max += Math.max(...q.opts.map(x => x.pts));

  });

  return { cur, proj, cats };

}

function grade(score) {

  const p = score / MAX_PTS * 100;

  if (p >= 80) return { label: 'Strong', color: '#2d7a4f', bg: '#e8f5ee' };

  if (p >= 35) return { label: 'Emerging', color: '#b07a3a', bg: '#fdf1e4' };

  return { label: 'Starting Out', color: '#888', bg: '#f0f0f0' };

}

    const gain = o.proj - o.pts;

    if (o.pts >= max * 0.75 || gain === 0) return null;

    return { name: q.cat, gain, desc: q.gapDesc };

  }).filter(Boolean).sort((a, b) => b.gain - a.gain).slice(0, 4);

}

const finalChallenges = [

  'People don\'t know we exist',

  'Visitors don\'t convert to donors',

  'Volunteer recruitment online',

  'Grant funders can\'t find us',

  'All of the above'

];

function sendToGoogleSheets(scores, gaps) {

  if (!GOOGLE_SCRIPT_URL || GOOGLE_SCRIPT_URL === 'YOUR_GOOGLE_APPS_SCRIPT_URL_HERE') return;

  const payload = {

    timestamp: new Date().toISOString(),

    name: contact.name,

    org: contact.org,

    email: contact.email,

    currentScore: scores.cur,

    projectedScore: scores.proj,

    maxScore: MAX_PTS,

    currentPct: pct(scores.cur, MAX_PTS),

    projectedPct: pct(scores.proj, MAX_PTS),

    grade: grade(scores.cur).label,

    biggestChallenge: finalChallenges[answers[questions.length]] || '',

    topGaps: gaps.map(g => g.name).join(', '),

    googleSearch: scores.cats['Google Search Presence'] ? scores.cats['Google Search Presence'].cur + '/' + scores.cats['Google Search Presence'].max : '',

    googleBusiness: scores.cats['Google Business Profile'] ? scores.cats['Google Business Profile'].cur + '/' + scores.cats['Google Business Profile'].max : '',

    websiteTraffic: scores.cats['Website Traffic'] ? scores.cats['Website Traffic'].cur + '/' + scores.cats['Website Traffic'].max : '',

    contentSEO: scores.cats['Content & SEO'] ? scores.cats['Content & SEO'].cur + '/' + scores.cats['Content & SEO'].max : '',

    googleAdsGrant: scores.cats['Google Ads Grant'] ? scores.cats['Google Ads Grant'].cur + '/' + scores.cats['Google Ads Grant'].max : '',

    aiVisibility: scores.cats['AI & Answer Engine Visibility'] ? scores.cats['AI & Answer Engine Visibility'].cur + '/' + scores.cats['AI & Answer Engine Visibility'].max : '',

    fundraisingPage: scores.cats['Fundraising Landing Page'] ? scores.cats['Fundraising Landing Page'].cur + '/' + scores.cats['Fundraising Landing Page'].max : ''

  };

  fetch(GOOGLE_SCRIPT_URL, {

    method: 'POST',

    mode: 'no-cors',

    headers: { 'Content-Type': 'application/json' },

    body: JSON.stringify(payload)

  }).catch(() => {});

}
